    let chartTemp, chartRain;

    async function loadWeather(lat, lon) {
      try {
        const url =
          `https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}` +
          `&current_weather=true` +
          `&hourly=temperature_2m,precipitation` +
          `&daily=temperature_2m_max,temperature_2m_min,precipitation_sum,weathercode` +
          `&timezone=Europe/Berlin`;

        const response = await fetch(url);
        if (!response.ok) throw new Error("Fehler beim Abrufen der Wetterdaten");
        const data = await response.json();

        const current = data.current_weather;
        document.getElementById("temp").innerText = Math.round(current.temperature) + "¬∞C";
        document.getElementById("desc").innerText = weatherText(current.weathercode);
        const currentIcon = document.getElementById("currentIcon");
        currentIcon.innerText = iconForCode(current.weathercode);
        currentIcon.setAttribute("aria-label", weatherText(current.weathercode));

        const forecastDiv = document.getElementById("forecast");
        forecastDiv.innerHTML = data.daily.time.slice(0, 3).map((day, i) => `
          <div class="card">
            <div class="icon" aria-label="${weatherText(data.daily.weathercode[i])}">${iconForCode(data.daily.weathercode[i])}</div>
            <h3>${new Date(day).toLocaleDateString("de-CH", { weekday:"short", day:"numeric", month:"numeric" })}</h3>
            <div class="temp">${Math.round(data.daily.temperature_2m_max[i])}¬∞ / ${Math.round(data.daily.temperature_2m_min[i])}¬∞</div>
            <div class="condition">Regen: ${data.daily.precipitation_sum[i].toFixed(1)} mm</div>
          </div>
        `).join("");

        const labels = data.hourly.time.slice(0, 24).map(t => new Date(t).getHours() + ":00");

        if (!chartTemp) {
          chartTemp = new Chart(document.getElementById("tempChart"), {
            type: 'line',
            data: { labels, datasets: [{ label: 'Temperatur (¬∞C)', data: data.hourly.temperature_2m.slice(0, 24), borderColor: 'var(--color-primary)', backgroundColor: 'rgba(59,130,246,0.2)', fill: true, tension: 0.3 }] },
            options: { responsive:true, plugins:{legend:{display:false}} }
          });
        } else {
          chartTemp.data.labels = labels;
          chartTemp.data.datasets[0].data = data.hourly.temperature_2m.slice(0, 24);
          chartTemp.update();
        }

        if (!chartRain) {
          chartRain = new Chart(document.getElementById("rainChart"), {
            type: 'bar',
            data: { labels, datasets: [{ label: 'Niederschlag (mm)', data: data.hourly.precipitation.slice(0, 24), backgroundColor: 'var(--color-rain)' }] },
            options: { responsive:true, plugins:{legend:{display:false}} }
          });
        } else {
          chartRain.data.labels = labels;
          chartRain.data.datasets[0].data = data.hourly.precipitation.slice(0, 24);
          chartRain.update();
        }

      } catch (error) {
        console.error(error);
        document.getElementById("temp").innerText = "--¬∞C";
        document.getElementById("desc").innerText = "Keine Daten";
      }
    }
